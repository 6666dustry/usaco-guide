from itertools import groupby

for _ in range(int(input())):
	string = input()

	b_amts = 0
	a_runs = []
	for char, amt in groupby(string):
		if char == "A":
			a_runs.append(len(list(amt)))
		else:
			b_amts += min(len(list(amt)), 2)

	if b_amts < len(a_runs):
		print(sum(a_runs) - min(a_runs))
	else:
		print(sum(a_runs))
